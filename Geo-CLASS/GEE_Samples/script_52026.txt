/**** Start of imports. If edited, may not auto-convert in the playground. ****/
var roi = ee.FeatureCollection("users/bkkumarbittu371/Andhra_Pradesht");
/***** End of imports. If edited, may not auto-convert in the playground. *****/
//JANUARY

var collection = ee.ImageCollection('COPERNICUS/S5P/OFFL/L3_O3')

  .select('O3_column_number_density')

  .filterBounds(roi)

  .filterDate('2019-01-01', '2019-01-31');



// This automatically pans the map to the middle of our area of interest

Map.centerObject(roi);



var datasets=collection.mean();

var clipped=datasets.clip(roi);



var Mean_jan_2019 = clipped.reduceRegion({

  reducer:ee.Reducer.mean(),

  geometry:roi,

  scale:500,

})

print ("Mean_jan_2019",Mean_jan_2019)
